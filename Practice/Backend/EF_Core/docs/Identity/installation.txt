Identity Installation

 1. Install package Microsoft.AspNetCore.Identity.EntityFrameworkCore
 2. IdentityDbContext class
  - In the Data folder, create a new class named AuthDbContext.cs
  - AuthDbContext class is responsible for creating new roles, users, and seeding all 
    that data into the database.

        using Microsoft.AspNetCore.Identity;
        using Microsoft.AspNetCore.Identity.EntityFrameworkCore;
        using Microsoft.EntityFrameworkCore;

        // Ignore Spelling: Bloggie

        namespace Bloggie.Data
        {
            public class AuthDbContext : IdentityDbContext
            {
                public AuthDbContext(DbContextOptions options) : base(options)
                {

                }

                // In order to seed the data, we override OnModelCreating method, which takes the builder, and pass it to the base class.
                protected override void OnModelCreating(ModelBuilder builder)
                {
                    base.OnModelCreating(builder);

                    var superAdminRoleId = "b67a3bfa-eb80-42c2-b096-43925edaec5d";
                    var adminRoleId = "9da61983-fbb8-48f2-8dd8-70ab98123f23";
                    var userRoleId = "28e8d689-8ea8-4db0-afd0-4c84f5220f21";

                    // Seed Roles (User, Admin, SuperAdmin)

                    // Creating a list of Roles
                    var roles = new List<IdentityRole>
                    {
                        new IdentityRole
                        {
                            Id = superAdminRoleId,
                            Name ="SuperAdmin",
                            NormalizedName = "SUPERADMIN",
                            ConcurrencyStamp = superAdminRoleId
                        },
                        new IdentityRole
                        {
                            Id = adminRoleId,
                            Name = "Admin",
                            NormalizedName = "ADMIN",
                            ConcurrencyStamp = adminRoleId
                        },
                        new IdentityRole
                        {
                            Id = userRoleId,
                            Name = "User",
                            NormalizedName = "USER",
                            ConcurrencyStamp = userRoleId
                        }

                    };

                    // Inserting the list of Roles data into the builder object (into the database).
                    // So when we run EF Core migration, these Roles will be taken as seeds, and inserted into the database
                    builder.Entity<IdentityRole>().HasData(roles);


                    // Seed/create a SuperAdminUser

                    // Create an Identity superAdminUser
                    var superAdminId = "e6a5af26-87be-427b-8a74-0cb43a2497b1";

                    var superAdminUser = new IdentityUser
                    {
                        Id = superAdminId,
                        UserName = "superadmin@bloggie.com",
                        NormalizedUserName = "SUPERADMIN@BLOGGIE.COM",
                        Email = "superadmin@bloggie.com",
                        NormalizedEmail = "SUPERADMIN@BLOGGIE.COM",
                        ConcurrencyStamp = superAdminId,
                    };

                    // Hashing superAdmin password
                    superAdminUser.PasswordHash = new PasswordHasher<IdentityUser>()
                    .HashPassword(user: superAdminUser, password: "SuperAdmin@123");

                    // Insert and seed superAdminUser into the database
                    builder.Entity<IdentityUser>().HasData(superAdminUser);

                    // Add all roles to SuperAdminUser, using  IdentityUserRole.
                    // IdentityUserRole requires a key of type string.

                    // By adding all roles to SuperAdminUser, they can log as User, Admin, and SuperAdminUser

                    var superAdminRoles = new List<IdentityUserRole<string>>
                    {
                    new IdentityUserRole<string>
                    {
                    RoleId = superAdminRoleId ,
                    UserId = superAdminId
                    },
                    new IdentityUserRole<string>
                    {
                        RoleId = adminRoleId,
                        UserId = superAdminId
                    },
                    new IdentityUserRole<string>
                    {
                        RoleId = userRoleId,
                        UserId = superAdminId,
                    }
                    };

                    // Seed and add superAdmin User data into the database
                    builder.Entity<IdentityUserRole<string>>().HasData(superAdminRoles);
                }
            }
        }

 3. Create a connection string in appsettings.json

    "BloggieAuthDb": "Server=TareqPC\\MSSQLSRV; Database=BloggieAuthDb; 
                    Trusted_Connection=True; TrustServerCertificate=True;"
    
 4. Inject the BloggieAuthDb connection string in Program.Cs
    
    Inject DbContext - Identity connection 
    var authConnectionString = builder.Configuration.GetConnectionString("BloogieAuthDb");
    builder.Services.AddDbContext<AuthDbContext>(options =>
    options.UseSqlServer(authConnectionString)
    );
    builder.Services.AddIdentity<IdentityUser, IdentityRole>()
        .AddEntityFrameworkStores<AuthDbContext>();

    Authenticate the Identity users to enable the Identity service. 
        app.UseAuthentication();    
       * We must insert this line BEFORE app.UseAuthorization(); 

 5. Run EF Core migration
    - Using Package Manager Console (PM), we migrate Identity 
      and create the new database BloggieAuthDb, with it's custom tables, column, rows, 
      and data.
    1.  PM>Add-Migration "Creating Auth Db" -Context AuthDbContext
    2. PM>Update-Database -Context AuthDbContext
