Database Context DbContext
 - Abstraction of the DB structure in code
 - Serves as a connection/bridge between entities and the database.
 - Lists the models and their DB table names
 - Instantiates a DB connection during app runtime.
 - Allows for on-creation customizations and custom table configurations.

 - Through DbContext, we can have the following features:
  - Querying (interacting with the database)
  - Performing CRUD operation on the database using EF Core
  - Change Tracking (migrations) 
  - Persistence
  - Manage Relationships
  - Caching
  - Transaction
  - Object Materialization

  using Microsoft.EntityFrameWorkCore;
  using EntityFrameWorkCore.Date;
  using EntityFrameWorkCore.Domain;

    class FootballLeagueDbContext: DbContext 
    {
      // MAKE SURE entities classes have public access modifier, otherwise
      // we cannot used them as entity types like in the example below.
      // Entities
      public DbSet<Team> Teams {get; set;}
      public DbSet<Coach> Coaches {get; set;}

     

      // Conn string
      protected override void onConfiguring(DbContextOptionsBuilder optionsBuilder)
      {
        optionsBuilder.UseSqlServer("
        Server=localHost\\MSSQLSRV; 
        Database="DB_Name";
        Integrated Security=True;       // Windows login
        TrustServerCertificate=True;
        Encrypt=False
        
        );
      }

    }
